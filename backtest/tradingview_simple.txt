// This Pine Script has been written for the newest Pine v6 syntax
//@version=6
strategy("Simple MA Cross", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=10)

// Inputs
fastLen = input.int(20, "Fast MA")
slowLen = input.int(50, "Slow MA")
src = input.close("Source")

// Calculate
fastMA = ta.sma(src, fastLen)
slowMA = ta.sma(src, slowLen)

// Plot
plot(fastMA, color=color.blue, title="Fast")
plot(slowMA, color=color.orange, title="Slow")

// Signals
longCondition = ta.crossover(fastMA, slowMA)
shortCondition = ta.crossunder(fastMA, slowMA)

// Execute
if longCondition
    strategy.entry("Long", strategy.long)

if shortCondition
    strategy.entry("Short", strategy.short)

// Close on opposite signal
if longCondition[1]
    strategy.close("Short")

if shortCondition[1]
    strategy.close("Long")
